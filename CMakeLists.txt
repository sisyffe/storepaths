cmake_minimum_required(VERSION 3.20)

# Standard
set(CMAKE_C_STANDARD 23)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Target

        ${CMAKE_CURRENT_SOURCE_DIR}/include
)

    ${CMAKE_CURRENT_SOURCE_DIR}/src/cfgpath.impl.osx.c
)
